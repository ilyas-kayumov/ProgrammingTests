string dayOfWeek = 2;
switch (dayOfWeek)
{
    case 0:
        Console.Write("Monday");
        break;
    case 1:
        Console.Write("Tuesday");
        break;
    case 2:
        Console.Write("Wednesday");
        break;
    case 3:
        Console.Write("Thursday");
        break;
    case 4:
        Console.Write("Friday");
        break;
}
