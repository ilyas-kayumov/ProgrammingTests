string code = "";
string country = "Unknown";

switch (country)
{
    case "USA":
        code = "en-us";
        break;
    case "UK":
        code = "en-uk";
        break;
    case "Russia":
        code = "ru-ru";
        break;
    default:
        code = "en-us";
        break;
}

Console.Write(code);
