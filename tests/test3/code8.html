string character = "";
string characterClass = "Mage";
bool isEvilCharacter = true;

if (isEvilCharacter)
{
    if (characterClass == "Mage")
    {
        character = "Witch";
    }
    else
    {
        character = "Skeleton";
    }
}
else
{
    if (characterClass == "Mage")
    {
        character = "White Mage";
    }
    else
    {
        character = "White Knight";
    }
}

Console.Write(character);
